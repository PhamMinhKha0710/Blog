class BlogContent {
    constructor() {
        this.topics = new Set();
        this.qualityStandard = "high";
    }
    
    addTopic(topic) {
        this.topics.add(topic);
        console.log(`Added new topic: ${topic}`);
        return this;
    }
    
    publishRegularly() {
        return this.topics.size > 0 
            ? "New content will be updated weekly!"

    }
}
