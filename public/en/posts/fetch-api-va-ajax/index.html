What is AJAX?

AJAX (Asynchronous JavaScript and XML) is a web development technique that allows sending and receiving data from a server without reloading the page. AJAX combines:

  • JavaScript and DOM to display and interact with data
  • XMLHttpRequest or Fetch API to exchange data asynchronously with the server
  • HTML, CSS to display and style information
  • XML, JSON or text to transfer data (JSON is most commonly used today)

AJAX revolutionized the web experience by allowing partial page updates instead of reloading the entire page, creating smoother and faster web applications.

XMLHttpRequest: The Foundation of AJAX

XMLHttpRequest (XHR) was the first JavaScript object used to make AJAX requests.

Basic Example with XMLHttpRequest

// Create XMLHttpRequest object
const xhr = new XMLHttpRequest();

// Configure request
xhr.open('GET', 'https://api.example.com/data', true);

// Set up handler for state changes
xhr.onreadystatechange = function() {
    // readyState 4 means request is complete
    if (xhr.readyState === 4) {
        if (xhr.status === 200) {
            // Process returned data
            const data = JSON.parse(xhr.responseText);
            console.log('Data received:', data);
        } else {
            console.error('HTTP error:', xhr.status);
        }
    }
};

// Send request
xhr.send();

Error Handling and Events in XMLHttpRequest

const xhr = new XMLHttpRequest();
xhr.open('GET', 'https://api.example.com/data', true);

// Use event handlers
xhr.onload = function() {
    if (xhr.status === 200) {
        const data = JSON.parse(xhr.responseText);
        console.log('Data received:', data);
    } else {
        console.error('HTTP error:', xhr.status);
    }
};

xhr.onerror = function() {
    console.error('Network connection error');
};

xhr.ontimeout = function() {
    console.error('Request timed out');
};

// Set 5 second timeout
xhr.timeout = 5000;

xhr.send();

POST Request with XMLHttpRequest

const xhr = new XMLHttpRequest();
xhr.open('POST', 'https://api.example.com/users', true);
xhr.setRequestHeader('Content-Type', 'application/json');

xhr.onload = function() {
    if (xhr.status === 201) {
        const createdUser = JSON.parse(xhr.responseText);
        console.log('User created:', createdUser);
    } else {
        console.error('Error creating user:', xhr.status);
    }
};

const userData = {
    name: 'John Doe',
    email: 'johndoe@example.com',
    age: 30
};

xhr.send(JSON.stringify(userData));

Fetch API: The Modern Approach

Fetch API is a modern JavaScript interface based on Promises for making HTTP requests. It simplifies the process of sending requests and handling responses compared to XMLHttpRequest.

GET Request with Fetch API

fetch('https://api.example.com/data')
    .then(response => {
        // Check if response is OK
        if (!response.ok) {
            throw new Error(`HTTP error! Status: ${response.status}`);
        }
        // Parse JSON response
        return response.json();
    })
    .then(data => {
        console.log('Data received:', data);
    })
    .catch(error => {
        console.error('Error fetching data:', error);
    });

POST Request with Fetch API

const userData = {
    name: 'Jane Smith',
    email: 'janesmith@example.com',
    age: 25
};

fetch('https://api.example.com/users', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json'
    },
    body: JSON.stringify(userData)
})
.then(response => {
    if (!response.ok) {
        throw new Error(`HTTP error! Status: ${response.status}`);
    }
    return response.json();
})
.then(newUser => {
    console.log('User created:', newUser);
})
.catch(error => {
    console.error('Error creating user:', error);
});

Fetch API Options

fetch('https://api.example.com/data', {
    method: 'GET', // GET, POST, PUT, DELETE, etc.
    headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
    },
    credentials: 'include', // include, same-origin, omit
    cache: 'no-cache', // default, no-cache, reload, force-cache, only-if-cached
    redirect: 'follow', // follow, error, manual
    referrerPolicy: 'no-referrer', // no-referrer, client
    mode: 'cors', // cors, no-cors, same-origin
    timeout: 5000 // NOT part of fetch API, requires AbortController
})
.then(response => response.json())
.then(data => console.log(data))
.catch(error => console.error('Error:', error));

Handling Timeout with Fetch and AbortController

Fetch API doesn’t directly support timeout, but you can use AbortController:

// Create AbortController to cancel fetch
const controller = new AbortController();
const signal = controller.signal;

// Set timeout
const timeout = setTimeout(() => {
    controller.abort();
}, 5000);

fetch('https://api.example.com/data', { signal })
    .then(response => {
        // Clear timeout since request completed
        clearTimeout(timeout);
        
        if (!response.ok) {
            throw new Error(`HTTP error! Status: ${response.status}`);
        }
        return response.json();
    })
    .then(data => {
        console.log('Data received:', data);
    })
    .catch(error => {
        if (error.name === 'AbortError') {
            console.error('Request timed out');
        } else {
            console.error('Fetch error:', error);
        }
    });

Async/Await with Fetch API

Async/await is modern syntax that makes working with Promises easier, including Fetch API.

async function fetchData() {
    try {
        const response = await fetch('https://api.example.com/data');
        
        if (!response.ok) {
            throw new Error(`HTTP error! Status: ${response.status}`);
        }
        
        const data = await response.json();
        console.log('Data received:', data);
        return data;
    } catch (error) {
        console.error('Error fetching data:', error);
    }
}

// Call the function
fetchData().then(data => {
    // Do something with data
});

Multiple Concurrent Requests with Promise.all

async function fetchMultipleResources() {
    try {
        const userPromise = fetch('https://api.example.com/users').then(r => r.json());
        const postsPromise = fetch('https://api.example.com/posts').then(r => r.json());
        const commentsPromise = fetch('https://api.example.com/comments').then(r => r.json());
        
        // Wait for all requests to complete
        const [users, posts, comments] = await Promise.all([
            userPromise, postsPromise, commentsPromise
        ]);
        
        console.log('Users:', users);
        console.log('Posts:', posts);
        console.log('Comments:', comments);
        
        return { users, posts, comments };
    } catch (error) {
        console.error('Error fetching multiple resources:', error);
    }
}

Handling Sequential Dependent Requests

async function fetchSequentialData() {
    try {
        // Get user list
        const usersResponse = await fetch('https://api.example.com/users');
        const users = await usersResponse.json();
        
        // Get details of first user
        const userId = users[0].id;
        const userDetailResponse = await fetch(`https://api.example.com/users/${userId}`);
        const userDetail = await userDetailResponse.json();
        
        // Get that user's posts
        const userPostsResponse = await fetch(`https://api.example.com/users/${userId}/posts`);
        const userPosts = await userPostsResponse.json();
        
        return {
            user: userDetail,
            posts: userPosts
        };
    } catch (error) {
        console.error('Error fetching sequential data:', error);
    }
}

Building an API Service Class

A modern approach is to build an API Service class to encapsulate and reuse API call logic:

class ApiService {
    constructor(baseUrl) {
        this.baseUrl = baseUrl;
        this.defaultHeaders = {
            'Content-Type': 'application/json',
            'Accept': 'application/json'
        };
    }
    
    // Set authentication token
    setAuthToken(token) {
        this.authToken = token;
    }
    
    // Create headers for request
    createHeaders(customHeaders = {}) {
        const headers = { ...this.defaultHeaders, ...customHeaders };
        
        if (this.authToken) {
            headers['Authorization'] = `Bearer ${this.authToken}`;
        }
        
        return headers;
    }
    
    // Common request method
    async request(endpoint, options = {}) {
        const url = this.baseUrl + endpoint;
        
        const fetchOptions = {
            method: options.method || 'GET',
            headers: this.createHeaders(options.headers),
            ...options
        };
        
        if (options.body && typeof options.body === 'object') {
            fetchOptions.body = JSON.stringify(options.body);
        }
        
        try {
            const response = await fetch(url, fetchOptions);
            
            // Handle response by data type
            let data;
            const contentType = response.headers.get('content-type');
            
            if (contentType && contentType.includes('application/json')) {
                data = await response.json();
            } else {
                data = await response.text();
            }
            
            // Handle HTTP errors
            if (!response.ok) {
                throw {
                    status: response.status,
                    statusText: response.statusText,
                    data
                };
            }
            
            return data;
        } catch (error) {
            console.error('API request failed:', error);
            throw error;
        }
    }
    
    // Utility methods
    async get(endpoint, options = {}) {
        return this.request(endpoint, { ...options, method: 'GET' });
    }
    
    async post(endpoint, body, options = {}) {
        return this.request(endpoint, { ...options, method: 'POST', body });
    }
    
    async put(endpoint, body, options = {}) {
        return this.request(endpoint, { ...options, method: 'PUT', body });
    }
    
    async patch(endpoint, body, options = {}) {
        return this.request(endpoint, { ...options, method: 'PATCH', body });
    }
    
    async delete(endpoint, options = {}) {
        return this.request(endpoint, { ...options, method: 'DELETE' });
    }
}

// Use the ApiService class
const api = new ApiService('https://api.example.com');

// Set token if user is logged in
api.setAuthToken(localStorage.getItem('authToken'));

// Use to call API
async function fetchUsers() {
    try {
        const users = await api.get('/users');
        console.log('Users:', users);
        return users;
    } catch (error) {
        console.error('Error fetching users:', error);
        throw error;
    }
}

// Create new user
async function createUser(userData) {
    try {
        const newUser = await api.post('/users', userData);
        console.log('Created user:', newUser);
        return newUser;
    } catch (error) {
        console.error('Error creating user:', error);
        throw error;
    }
}

Handling Errors and Loading State

In real applications, handling loading state and errors is crucial:

// Assume we have the following DOM elements:
// <div id="users-container"></div>
// <div id="loading-indicator" class="hidden">Loading...</div>
// <div id="error-message" class="hidden"></div>

const usersContainer = document.getElementById('users-container');
const loadingIndicator = document.getElementById('loading-indicator');
const errorMessage = document.getElementById('error-message');

async function loadUsers() {
    // Show loading indicator
    loadingIndicator.classList.remove('hidden');
    // Hide previous error message if any
    errorMessage.classList.add('hidden');
    // Clear old content
    usersContainer.innerHTML = '';
    
    try {
        const users = await api.get('/users');
        
        // Display user list
        users.forEach(user => {
            const userElement = document.createElement('div');
            userElement.classList.add('user-card');
            userElement.innerHTML = `
                <h3>${user.name}</h3>
                <p>Email: ${user.email}</p>
                <p>Phone: ${user.phone}</p>
            `;
            usersContainer.appendChild(userElement);
        });
    } catch (error) {
        // Display error message
        errorMessage.textContent = `Error: ${error.status === 401 
            ? 'You do not have access' 
            : 'Unable to load user data'}`;
        errorMessage.classList.remove('hidden');
    } finally {
        // Hide loading indicator
        loadingIndicator.classList.add('hidden');
    }
}

// Call function when page loads
document.addEventListener('DOMContentLoaded', loadUsers);

        const months = ['January', 'February', 'March', 'April', 'May', 'June', 
                        'July', 'August', 'September', 'October', 'November', 'December'];
        
        return `${days[date.getDay()]}, ${months[date.getMonth()]} ${date.getDate()}, ${date.getFullYear()}`;

            const response = await fetch(

                throw new Error('City not found');
            }
            
            const data = await response.json();
            
            // Update UI with weather data

    .then(data => {

3. Add Timeout to Requests

const fetchWithTimeout = (url, options = {}, timeout = 5000) => {
    return Promise.race([
        fetch(url, options),
        new Promise((_, reject) => 
            setTimeout(() => reject(new Error('Request timed out')), timeout)
        )
    ]);
};

4. Reuse Fetch Logic

// Create a helper function

5. Debounce Search

// Debounce function
function debounce(func, delay) {
    let timeout;
    return function() {
        const context = this;
        const args = arguments;
        clearTimeout(timeout);
        timeout = setTimeout(() => func.apply(context, args), delay);
    };
}

    const query = event.target.value;

        fetchSearchResults(query);
    }
}, 300);
